```python
_THREE_WAY_SUM_MIN = 0.95
_THREE_WAY_SUM_MAX = 1.05
_THREE_WAY_SPORTS = frozenset({"soccer", "rugby", "afl", "handball"})


def _passes_three_way_sum_filter(markets: list[MarketData], event_id: str) -> bool:
    """SPEC-015: 3-way sport için event'teki market'lerin yes_price toplamı 0.95-1.05.

    2-way sporlar (event başına tek market) → her zaman geçer.
    3-way sport + 3 market → toplam check.
    3-way sport + 2 market → sum check (eksik market henüz yayınlanmamış olabilir).
    """
    if not markets or not event_id:
        return True
    sport = (markets[0].sport_tag or "").lower()
    if not any(s in sport for s in _THREE_WAY_SPORTS):
        return True  # 2-way sport, filter atla
    event_markets = [m for m in markets if m.event_id == event_id]
    if len(event_markets) < 2:
        return True  # tek market, sum check anlamsız
    total = sum(m.yes_price for m in event_markets)
    return _THREE_WAY_SUM_MIN <= total <= _THREE_WAY_SUM_MAX
```

Scanner main loop'unda her event_id için bu filter'ı uygula. Reddedilenler `skipped` log'a düşer (`reason="not_three_way_h2h"`).

Konkret entegrasyon noktası `_passes_filters` veya pipeline post-filter aşamasında.

```python
"""Event grouper — Polymarket flat market list'inden event-bazlı grup üretir.

SPEC-015: 3-way soccer event'leri 3 ayrı binary market olarak gelir.
Bu modül event_id ile grupluyor + market_type sınıflandırması yapıyor.

2-way sporlar (MLB/NBA/...) tek market'lik group olarak pass-through olur.
"""
from __future__ import annotations

from collections import defaultdict
from dataclasses import dataclass

from src.models.market import MarketData

_THREE_WAY_SPORTS = frozenset({"soccer", "rugby", "afl", "handball"})

_DRAW_KEYWORDS = ("end in a draw", "draw", "level", "tie")


def _is_draw_market(question: str) -> bool:
    q = (question or "").lower()
    return any(kw in q for kw in _DRAW_KEYWORDS)


def _is_three_way_sport(sport_tag: str) -> bool:
    s = (sport_tag or "").lower()
    return any(tw in s for tw in _THREE_WAY_SPORTS)


@dataclass
class EventGroup:
    """Bir event için gruplanmış market'ler."""
    event_id: str
    market_type: str  # "BINARY" | "THREE_WAY"
    markets: list[MarketData]

    def classify_outcomes(self) -> tuple[MarketData | None, MarketData | None, MarketData | None]:
        """3-way için (home, draw, away) tuple. Question keyword based.

        Sıra: question'daki team isminin Polymarket pattern'inde önce gelen home,
        sonra draw, sonra away. Draw market 'draw'/'tie' keyword içerir.

        Returns: (home, draw, away). Eksik outcome None döner.
        """
        if self.market_type != "THREE_WAY":
            return (None, None, None)

        draw_market = next((m for m in self.markets if _is_draw_market(m.question)), None)
        non_draw = [m for m in self.markets if m is not draw_market]

        # Home/away ayırımı: ilk market'i home, sonrayı away kabul et
        # (Polymarket question pattern'inde "Will {Home} win?" önce gelir genelde)
        if len(non_draw) == 2:
            home, away = non_draw[0], non_draw[1]
        else:
            home = non_draw[0] if non_draw else None
            away = non_draw[1] if len(non_draw) > 1 else None

        return (home, draw_market, away)


def group_markets_by_event(markets: list[MarketData]) -> list[EventGroup]:
    """Flat MarketData listesi → event_id bazlı EventGroup listesi."""
    by_event: dict[str, list[MarketData]] = defaultdict(list)
    for m in markets:
        if not m.event_id:
            continue
        by_event[m.event_id].append(m)

    groups: list[EventGroup] = []
    for eid, ms in by_event.items():
        sport = ms[0].sport_tag if ms else ""
        if _is_three_way_sport(sport) and len(ms) >= 2:
            mtype = "THREE_WAY"
        else:
            mtype = "BINARY"
        groups.append(EventGroup(event_id=eid, market_type=mtype, markets=ms))

    return groups
```

```python
def _bm(prob_home: float, prob_draw: float, prob_away: float, conf: str = "A") -> dict:

        "home": BookmakerProbability(probability=prob_home, confidence=conf,
            bookmaker_prob=prob_home, num_bookmakers=10.0, has_sharp=True),
        "draw": BookmakerProbability(probability=prob_draw, confidence=conf,
```

```python
"""3-way entry strategy — Soccer/Rugby/AFL/Handball (SPEC-015).

Bookmaker'ın 3 outcome arasında en yüksek olasılığa sahip outcome'u (favori) seçer.
Filter: absolute prob >= favorite_threshold AND margin to second highest >= favorite_margin.
Edge eşiği: |bookmaker_prob - market_yes_price| >= min_edge.
"""
from __future__ import annotations

from dataclasses import dataclass

from src.domain.analysis.probability import BookmakerProbability
from src.models.enums import Direction, EntryReason
from src.models.market import MarketData
from src.models.signal import Signal


@dataclass
class ThreeWaySelection:
    """Hangi outcome seçildi + gerekçe."""
    outcome: str  # "home" | "draw" | "away"
    market: MarketData
    bm_prob: BookmakerProbability


def _select_favorite(probs: dict[str, BookmakerProbability]) -> tuple[str, float] | None:
    """En yüksek bookmaker_prob outcome'u + olasılık. Eşitlik → None (tie-break)."""
    sorted_outcomes = sorted(probs.items(), key=lambda kv: -kv[1].probability)
    top_outcome, top_prob_obj = sorted_outcomes[0]
    second_prob_obj = sorted_outcomes[1][1] if len(sorted_outcomes) > 1 else None
    if second_prob_obj is not None and abs(top_prob_obj.probability - second_prob_obj.probability) < 1e-9:
        return None  # tie
    return top_outcome, top_prob_obj.probability


def _passes_favorite_filter(probs: dict[str, BookmakerProbability], favorite_outcome: str,
                            threshold: float, margin: float) -> bool:
    """Absolute >= threshold AND margin to second highest >= margin."""
    fav_prob = probs[favorite_outcome].probability
    if fav_prob < threshold:
        return False
    others = [p.probability for k, p in probs.items() if k != favorite_outcome]
    second_highest = max(others) if others else 0.0
    return (fav_prob - second_highest) >= margin


def evaluate(
    home_market: MarketData,
    draw_market: MarketData,
    away_market: MarketData,
    probs: dict[str, BookmakerProbability],
    min_edge: float = 0.06,
    favorite_threshold: float = 0.40,
    favorite_margin: float = 0.07,
) -> Signal | None:
    """3-way entry kararı. None → koşul sağlanmadı."""
    # 1. C-conf skip
    if any(p.confidence == "C" for p in probs.values()):
        return None

    # 2. Favori seçimi (tie-break: skip)
    selection = _select_favorite(probs)
    if selection is None:
        return None
    fav_outcome, fav_prob = selection

    # 3. Favorite filter (absolute + margin)
    if not _passes_favorite_filter(probs, fav_outcome, favorite_threshold, favorite_margin):
        return None

    # 4. Hangi market'e gireceğiz?
    fav_market_map = {"home": home_market, "draw": draw_market, "away": away_market}
    fav_market = fav_market_map[fav_outcome]
    if fav_market is None:
        return None  # outcome eksik (3-way grouping inkomplet)

    # 5. Edge hesabı (BUY_YES on the favorite market)
    market_yes = fav_market.yes_price
    edge = fav_prob - market_yes
    if edge < min_edge:
        return None

    return Signal(
        condition_id=fav_market.condition_id,
        direction=Direction.BUY_YES,  # her outcome'un kendi market'ında BUY_YES
        anchor_probability=fav_prob,
        market_price=market_yes,
        edge=edge,
        confidence=probs[fav_outcome].confidence,
        size_usdc=0.0,  # gate sizing uygular
        entry_reason=EntryReason.NORMAL,  # ya da YENİ EntryReason.THREE_WAY
        bookmaker_prob=fav_prob,
        num_bookmakers=probs[fav_outcome].num_bookmakers,
        has_sharp=probs[fav_outcome].has_sharp,
        sport_tag=fav_market.sport_tag,
        event_id=fav_market.event_id or "",
    )
```

```python
"""Soccer score exit (SPEC-015).

HOME/AWAY position:
- 0-65' HOLD (comeback potential)
- 65'+ 2 gol geride → EXIT
- 75'+ 1 gol geride → EXIT

DRAW position:
- 0-70' 0-0 → HOLD (draw değeri zirvede)
- 75'+ herhangi gol atıldı → EXIT
- Knockout maç + 90+stoppage → AUTO-EXIT (uzatma+pen draw'ı bitirir)

Sport config'ten okur (DRY — diğer 3-way sporlara uyarlanır).
"""
from __future__ import annotations

from dataclasses import dataclass

from src.config.sport_configs import get_sport_config
from src.models.enums import ExitReason


@dataclass
class SoccerExitResult:
    reason: ExitReason
    detail: str


def check(score_info: dict, sport_tag: str = "soccer") -> SoccerExitResult | None:
    if not score_info.get("available"):
        return None

    minute = score_info.get("minute")
    if not isinstance(minute, int) or minute < 0:
        return None

    state = score_info.get("regulation_state", "")
    if state == "pre" or state == "":
        return None

    cfg = get_sport_config(sport_tag) or {}
    if not cfg:
        return None  # config yoksa skip

    our_outcome = score_info.get("our_outcome", "")  # "home" | "draw" | "away"
    home_score = int(score_info.get("home_score", 0))
    away_score = int(score_info.get("away_score", 0))
    knockout = bool(score_info.get("knockout", False))

    # ── DRAW position ──
    if our_outcome == "draw":
        # Knockout 90+stoppage auto-exit
        if knockout and minute >= cfg.get("regulation_minutes", 90) and cfg.get("knockout_auto_exit_draw"):
            return SoccerExitResult(
                reason=ExitReason.SCORE_EXIT,
                detail=f"draw_knockout_auto: minute={minute}",
            )
        # Goal scored after draw_exit_after_goal threshold
        any_goal = home_score > 0 or away_score > 0
        if any_goal and minute >= cfg.get("draw_exit_after_goal", 75):
            return SoccerExitResult(
                reason=ExitReason.SCORE_EXIT,
                detail=f"draw_goal_late: minute={minute} score={home_score}-{away_score}",
            )
        return None

    # ── HOME/AWAY position ──
    if our_outcome not in ("home", "away"):
        return None  # bilinmeyen outcome

    # Deficit hesabı (bizim takımın ne kadar geride olduğu)
    if our_outcome == "home":
        deficit = away_score - home_score
    else:  # away
        deficit = home_score - away_score

    if deficit <= 0:
        return None  # önde veya berabere → HOLD

    first_half_lock = cfg.get("score_exit_first_half_lock", 65)
    if minute < first_half_lock:
        return None  # HOLD

    if minute >= cfg.get("score_exit_2goal_minute", 65) and deficit >= 2:
        return SoccerExitResult(
            reason=ExitReason.SCORE_EXIT,
            detail=f"2_goal_deficit: minute={minute} deficit={deficit}",
        )

    if minute >= cfg.get("score_exit_1goal_minute", 75) and deficit >= 1:
        return SoccerExitResult(
            reason=ExitReason.SCORE_EXIT,
            detail=f"1_goal_deficit: minute={minute} deficit={deficit}",
        )

    return None
```
